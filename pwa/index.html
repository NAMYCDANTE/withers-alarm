    function openCalendar(id) {

      const r = allData.find(x => x.id === id);

      if (!r) return;


      // 날짜 파싱 (2026-05-07, 2026.5.7 등)

      const dm = dateStr.match(/(\d{4})[.\-\/](\d{1,2})[.\-\/](\d{1,2})/);                                         

      if (!dm) { alert('날짜 형식을 인식할 수 없습니다.'); return; }                                               

      const year  = dm[1];                                                                                         

      const month = dm[2].padStart(2, '0');                                                                        

      const day   = dm[3].padStart(2, '0');                                                                        

                  

      // 시간 파싱 (오전 9시, 오후 2시 등)

      const tm = r.time.match(/(오전|오후)\s*(\d{1,2})시/);                                                        

      if (tm) {                                                                                                    

        hour = parseInt(tm[2]);

        if (tm[1] === '오후' && hour !== 12) hour += 12;                                                           

        if (tm[1] === '오전' && hour === 12) hour = 0;

      }

      const details = `고객명: ${r.customer_name}\n연락처: ${r.phone}\n에어컨: ${r.ac_info ||

  `${r.unit_count}대`}`;
